MODULE FILL_DYNNR OUTPUT.

    WHEN  'FC1'.
      SY-DYNNR = 501.
    WHEN  'FC2'.
      SY-DYNNR = 502.

      TAB1-ACTIVETAB = 'FC1'.
      SY-DYNNR = 501.
  ENDCASE.
ENDMODULE.

MODULE FILL_DATA OUTPUT.
  CLEAR :GS_LIST501, GS_LIST502.
  SELECT SINGLE *
 FROM ZTADDR_11
 WHERE PERNR = @GS_LIST-PERNR
 INTO  CORRESPONDING FIELDS OF @GS_LIST501.

  GS_LIST501-PERNR = GS_LIST-PERNR.

  SELECT SINGLE * FROM ZTCONT_11
  WHERE PERNR = @GS_LIST-PERNR
  INTO CORRESPONDING FIELDS OF @GS_LIST502.
  GS_LIST502-PERNR = GS_LIST-PERNR.
ENDMODULE.

MODULE SCREEN_INIT_500 OUTPUT.
  LOOP AT SCREEN.
    IF SCREEN-GROUP1 = 'G1' AND GV_DISP = 'X'.
      SCREEN-INPUT = 1.
      MODIFY SCREEN.
    ELSE.
      SCREEN-INPUT = 0.
      MODIFY SCREEN.
    ENDIF.
    ENDLOOP.
ENDMODULE.
